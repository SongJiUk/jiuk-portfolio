01퀘스트 상태 관리 시스템

퀘스트를 NotStarted / InProgress / Completed 세 가지 상태로 관리합니다. 조건 리스트를 순회해 모든 조건이 충족되면 자동으로 완료 처리됩니다.

C#QuestSystem.cs
public enum QuestState
{
    NotStarted, InProgress, Completed
}

public class QuestSystem
{
    public void CheckCompletion(Quest q)
    {
        if (q.State != QuestState.InProgress) return;

        bool done = q.Conditions.All(c => c.IsMet());
        if (done) q.State = QuestState.Completed;
    }
}
02ObjectPool — 몬스터 풀 관리

몬스터를 매 프레임 생성/파괴하는 대신 ObjectPool로 재사용해 GC 부하를 줄였습니다. OnCreate/OnGet/OnRelease/OnDestroy 콜백을 각각 구현해 상태를 명확히 관리합니다.

C#PoolManager.cs
public class PoolManager : MonoBehaviour
{
    [SerializeField] private Monster _prefab;
    ObjectPool<Monster> _pool;

    void Awake()
    {
        _pool = new ObjectPool<Monster>(
            createFunc:      () => Instantiate(_prefab),
            actionOnGet:     m  => m.gameObject.SetActive(true),
            actionOnRelease: m  => m.gameObject.SetActive(false),
            actionOnDestroy: m  => Destroy(m.gameObject),
            maxSize: 100
        );
    }

    public Monster Spawn(Vector3 pos)
    {
        Monster m = _pool.Get();
        m.transform.position = pos;
        return m;
    }

    public void Return(Monster m)
        => _pool.Release(m);
}
