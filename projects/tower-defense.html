01A* 경로탐색 + 동적 재탐색

A* 알고리즘을 직접 구현. 타워 설치 시 RecalculatePath() 이벤트로 모든 적이 현재 위치에서 즉시 경로를 재계산합니다. 경로가 완전히 막히면 설치를 취소해 적이 코어로 도달할 경로를 항상 보장합니다. 맨해튼 거리 휴리스틱, 탐색 후 G/H/Parent 초기화로 다음 호출 오염을 방지합니다.

C#PathFinder.cs
public List<Vector3> FindPath(Vector3 startPos, Vector3 endPos)
{
    GridNode startNode = Managers.Grid.GetNode(startPos);
    GridNode endNode   = Managers.Grid.GetNode(endPos);

    var open    = new List<GridNode>();
    var closed  = new HashSet<GridNode>();
    var visited = new List<GridNode>();

    startNode.G = 0;
    startNode.H = GetHeuristic(startNode, endNode); // 맨해튼 거리
    open.Add(startNode);

    while (open.Count > 0)
    {
        GridNode current = GetLowestF(open);
        if (current == endNode)
        {
            List<Vector3> path = TracePath(endNode);
            ResetNodes(visited); // G·H·Parent 초기화
            LastPath = path;
            return path;
        }

        open.Remove(current);
        closed.Add(current);

        foreach (var (dx, dz) in _directions) // 상하좌우 4방향
        {
            GridNode nb = grid.GetNode(current.GridX + dx, current.GridZ + dz);
            if (nb == null || !nb.CanWalk || closed.Contains(nb)) continue;
            int newG = current.G + 1;
            if (!open.Contains(nb)) { nb.G = newG; nb.H = GetHeuristic(nb, endNode); open.Add(nb); }
            else if (newG < nb.G) nb.G = newG;
        }
    }
    return null; // 경로 없음 → 타워 설치 취소
}
02버프/디버프 시스템 (BuffHandler)

IBuff 인터페이스 + BuffHandler 컴포넌트로 슬로우·독·동결·공격강화를 다중 스택 관리합니다. AllowStack 플래그로 중복 시 Refresh/스택 분기를 결정하고, StatModifier 패턴으로 여러 버프가 중첩된 최종 스탯을 연산합니다. ITickable로 등록해 별도 코루틴 없이 매 프레임 만료를 체크합니다.

C#BuffHandler.cs
public void AddEffect(BuffEffect effect)
{
    if (!effect.AllowStack)
    {
        foreach (var e in _effects)
        {
            if (e is BuffEffect existing && existing.EffectType == effect.EffectType)
            { existing.Refresh(); return; } // 중복 시 지속시간만 갱신
        }
    }
    _effects.Add(effect);
    effect.OnApply(this);
}

// ITickable.Tick: 만료 효과를 버퍼에 모아 한 번에 제거 (순회 중 컬렉션 수정 방지)
public void Tick(float dt)
{
    foreach (var e in _effects)
    { e.Tick(dt); if (e.IsExpired) _expiredBuffer.Add(e); }

    foreach (var e in _expiredBuffer)
    { e.OnRemove(this); _effects.Remove(e); }
    _expiredBuffer.Clear();
}

// StatModifier로 최종 스탯 계산 (배율·덧셈 조합)
public float GetStat(Define.StatType statType, float baseValue)
{
    foreach (var m in _modifiers)
        if (m.StatType == statType) baseValue = m.Apply(baseValue);
    return baseValue;
}
03EffectManager - Pool 기반 이펙트 일원화

투사체 착탄·사망·버프·스킬·타워 설치·번개 체인 등 모든 이펙트를 EffectManager 한 곳으로 통일했습니다. Play(key, pos)는 위치 이펙트, PlayLine(key, from, to)는 방향성 이펙트(번개 등)에 사용합니다. async void 대신 UniTaskVoid를 사용해 예외 안전성을 확보하고, duration 후 자동으로 풀에 반환합니다.

C#EffectManager.cs
public void Play(string key, Vector3 pos, float duration = 2f)
{
    GameObject go = Managers.PoolM.Pop(key);
    if (go == null) return;
    go.transform.position = pos;
    ReturnAsync(go, duration).Forget();
}

// 번개 체인처럼 시작→목표 방향으로 향하는 방향성 이펙트
public void PlayLine(string key, Vector3 from, Vector3 to, float dur = 0.3f)
{
    GameObject go = Managers.PoolM.Pop(key);
    if (go == null) return;
    go.transform.position = from;
    go.transform.rotation = Quaternion.LookRotation(to - from);
    ReturnAsync(go, dur).Forget();
}

private async UniTaskVoid ReturnAsync(GameObject go, float duration)
{
    await UniTask.Delay(TimeSpan.FromSeconds(duration));
    if (go != null && go.activeSelf) Managers.PoolM.Push(go);
}
04타워 6종 + 시너지 시스템

Basic·Cannon·Slow·Sniper·Poison·Lightning 각각 독립 Controller로 구현했습니다. 타워를 배치하면 SynergyManager가 조합을 감지해 OnSynergyChanged 이벤트를 발행하고, 모든 타워가 구독해 스탯을 자동 재계산합니다. 카드 효과도 OnCardApplied 이벤트로 동일하게 반영됩니다. 업그레이드 수치는 ScriptableObject에 분리해 코드 변경 없이 밸런싱이 가능합니다.

C#TowerController.cs
void OnEnable()
{
    // 카드 효과·시너지 변경 시 스탯 자동 재계산
    Managers.GameM.OnCardApplied     += ApplyStats;
    Managers.SynergyM.OnSynergyChanged += ApplyStats;
}

void OnDisable()
{
    Managers.GameM.OnCardApplied     -= ApplyStats;
    Managers.SynergyM.OnSynergyChanged -= ApplyStats;
    Managers.SynergyM.Unregister(Data.towerType);
}
05Managers 싱글톤 아키텍처

@Managers GameObject를 자동 생성 후 DontDestroyOnLoad로 유지합니다. Pool·Resource·Effect·UI·Game·Card·Skill·Synergy·Sound·Save·Firebase 등 15개 서브매니저를 모듈화해 씬 어디서든 Managers.PoolM, Managers.SkillM으로 접근합니다. 씬 전환 시 Managers.Clear()로 Pool·Resource·UI·Synergy를 일괄 초기화합니다.

C#Managers.cs
public class Managers : MonoBehaviour
{
    readonly PoolManager     poolManager     = new();
    readonly EffectManager   effectManager   = new();
    readonly SkillManager    skillManager    = new();
    readonly SynergyManager  synergyManager  = new();
    readonly SaveManager     saveManager     = new();
    // ... 15개 서브매니저

    void OnApplicationPause(bool pause)
    {
        if (pause) SaveM?.SyncToFirebase(); // 백그라운드 진입 시 자동 저장
    }

    public static void Clear()
    {
        PoolM.Clear(); ResourceM.Clear();
        UIM.Clear();  SynergyM.Clear();
    }
}
06로그라이크 카드 · 스킬 시스템

웨이브 클리어 후 카드 3장 중 1장을 선택합니다. 카드 카테고리는 A(전투강화) / B(경제) / C(특수) / D(스킬)로 구분됩니다. 스킬은 5종, 마스터 레벨 3, 최대 3개 보유·교체 가능합니다. 교체 시 레벨이 초기화됩니다. 모든 카드·스킬 데이터는 ScriptableObject로 분리해 하드코딩이 없습니다.

07Firebase 저장 연동

Firebase Authentication 익명 로그인 + Firestore 세이브 데이터 저장을 연동했습니다. SaveManager는 PlayerPrefs와 Firebase 이중 저장 구조이며, OnApplicationPause에서 앱 백그라운드 진입 시 자동으로 Firebase에 동기화합니다.

08업적 · 난이도 · 설정 시스템

AchievementManager로 업적 달성 시 팝업 알림을 표시합니다. 난이도는 쉬움·보통·어려움 3단계로 적 HP 배율을 적용합니다. 설정창은 BGM/SFX 슬라이더, 그래픽 품질(Low/Medium/High)을 제공하며 PlayerPrefs에 저장합니다.

🐛 트러블슈팅

Q. 타워 설치 후 적이 완전히 막혀 코어로 이동 불가
설치 전 A*로 경로를 미리 탐색해 경로가 null이면 설치를 취소합니다. 경로 재계산은 이벤트(RecalculatePath)로 모든 적에게 브로드캐스트해 각자 현재 위치에서 재탐색합니다.
Q. 웨이브 시작 버튼을 눌러도 적이 스폰되지 않음
Object_WaveAnnounce 오브젝트가 없어서 BeginSpawning()이 미호출되는 버그. OnWaveStart에서 직접 호출하도록 수정해 오브젝트 의존성을 제거했습니다.
Q. 스테이지 로드 시 stageKey 불일치로 StageData가 null
Addressables 키를 "Stage{n}"으로 등록했지만 코드에서 "Stage{n}Data"로 조회해 발생. 키 명명 규칙을 "Stage{n}Data"로 통일해 해결했습니다.
Q. UI_ForceClose 팝업이 이중으로 닫혀 Pool double-release 오류
_closed bool 플래그를 추가해 첫 Close 호출 이후 중복 호출을 무시하도록 처리했습니다.
