⚙️ 구현된 시스템

01ITickable / UpdateManager

개별 MonoBehaviour의 Update() 대신 중앙 UpdateManager가 ITickable 구현체를 일괄 호출합니다. toAdd / toRemove 버퍼를 사용해 이터레이션 중 컬렉션 수정 문제를 안전하게 처리합니다.

C#UpdateManager.cs
public interface ITickable
{
    void Tick(float _deltaTime);
}

public class UpdateManager : MonoBehaviour
{
    private readonly HashSet<ITickable> tickables = new();
    private readonly HashSet<ITickable> toAdd = new();
    private readonly HashSet<ITickable> toRemove = new();

    public void Register(ITickable _tickable)   => toAdd.Add(_tickable);
    public void UnRegister(ITickable _tickable) => toRemove.Add(_tickable);

    private void Update()
    {
        foreach (var t in toAdd) tickables.Add(t);
        toAdd.Clear();
        foreach (var t in toRemove) tickables.Remove(t);
        toRemove.Clear();

        foreach (var tick in tickables)
            tick.Tick(Time.deltaTime);
    }
}
