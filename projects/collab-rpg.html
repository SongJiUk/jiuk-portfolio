01싱글톤 매니저 베이스

GameManager, SceneManager, SoundManager 등 모든 매니저가 상속받는 공통 Singleton 베이스 클래스를 설계했습니다. DontDestroyOnLoad로 씬 전환 시에도 인스턴스를 유지합니다.

C#Singleton.cs
public class Singleton<T> : MonoBehaviour
    where T : MonoBehaviour
{
    static T _instance;

    public static T Instance
    {
        get
        {
            if (_instance == null)
                _instance = FindObjectOfType<T>();
            return _instance;
        }
    }

    protected virtual void Awake()
    {
        _instance = this as T;
        DontDestroyOnLoad(gameObject);
    }
}
